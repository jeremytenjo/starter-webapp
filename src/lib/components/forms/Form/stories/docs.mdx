# Form

If fields not directly insode Form use `useFormContext` to retrieve `register`

```js
import React from 'react'
import Box from '@mui/material/Box'
import Button from '@mui/material/Button'
import { useFormContext } from 'react-hook-form'

import Select from '../../../../../../../forms/fields/Select/Select'
import Form from '../../../../../../../forms/Form/Form'

type Props = { name?: string }

export default function Example({ name }: Props) {
  const seasonsList = [
    {
      label: 'Season 1',
      value: 1,
    },
  ]
  const episodeList = [
    {
      label: 'Episode 1',
      value: 1,
    },
  ]

  const onSubmit = (values) => {
    console.log(values)
  }

  return (
    <Form
      onSubmit={onSubmit}
      sx={{ display: 'grid', gridAutoFlow: 'row', gap: 2, width: '100%' }}
    >
      <FormFields seasonsList={seasonsList} episodeList={episodeList} />
    </Form>
  )
}

const FormFields = ({ seasonsList, episodeList }) => {
  const { register } = useFormContext()

  return (
    <>
      <Box
        sx={{
          display: 'grid',
          gridAutoFlow: 'column',
          justifyContent: 'space-between',
          gridTemplateColumns: '1fr 1fr',
          gap: 2,
        }}
      >
        <Select
          register={register}
          label='Season'
          name='selected_season'
          options={seasonsList}
        />
        <Select
          register={register}
          label='Episode'
          name='selected_episode'
          options={episodeList}
        />
      </Box>

      <Button
        sx={{
          color: 'white.main',
          width: '100%',
          display: 'block',
          margin: '0 auto',
        }}
        variant='contained'
        size='large'
        type='submit'
      >
        Play
      </Button>
    </>
  )
}
```
